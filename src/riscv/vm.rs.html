1//! # Virtual Machine (VM)
2//!
3//! This module implements the lowest-level abstraction of the emulator,
4//! modeling memory, registers, and basic I/O devices.
5//!
6//! The VM provides a safe and deterministic interface over raw machine
7//! state, which is used by higher layers to implement instruction execution.
8//!
9//! ## Responsibilities
10//!
11//! - Manage RAM and enforce memory safety
12//! - Provide access to general-purpose registers (`x0..x31`)
13//! - Maintain the program counter (PC)
14//! - Expose memory-mapped I/O (MMIO)
15//!
16//! ## Memory model
17//!
18//! Memory is represented as a contiguous byte array and follows a
19//! little-endian layout, as defined by the RISC-V specification.
20//!
21//! All memory accesses are bounds-checked. Invalid accesses result in
22//! [`VMError::MemoryOutOfBounds`] and do not modify the VM state.
23//!
24//! ## Registers
25//!
26//! The VM exposes 32 general-purpose registers (`x0..x31`).
27//!
28//! Register `x0` is hardwired to zero and ignores writes, as required by
29//! the RISC-V specification.
30//!
31//! ## Program counter (PC)
32//!
33//! The program counter stores the address of the next instruction to be
34//! executed.
35//!
36//! The VM ensures that the PC always points to a valid address in RAM,
37//! but does not enforce instruction alignment. Alignment is the
38//! responsibility of the CPU layer.
39//!
40//! ## Memory-mapped I/O
41//!
42//! A minimal UART device is exposed via memory-mapped registers:
43//!
44//! | Address        | Name   | Description              |
45//! |----------------|--------|--------------------------|
46//! | `0x1000_0000`  | TX     | Write a character        |
47//! | `0x1000_0000`  | RX     | Read a character (block) |
48//! | `0x1000_0004`  | STATUS | Transmitter status       |
49//!
50//! ### Behavior
51//!
52//! - Writing to TX prints a character to standard output
53//! - Reading from RX blocks until input is available
54//! - STATUS returns the transmitter readiness flag
55//!
56//! ## Architecture
57//!
58//! The VM is used by the [`crate::cpu`] layer, which is responsible for
59//! instruction decoding, execution, and enforcing ISA-level constraints
60//! such as alignment and control flow.
61//!
62//! The VM itself does not interpret instructions and should be considered
63//! a pure hardware abstraction layer.
64//!
65//! ## Error handling
66//!
67//! Errors are represented by [`VMError`] and correspond to invalid
68//! operations at the hardware level (e.g. out-of-bounds memory access).
69//!
70//! Higher layers wrap these errors to provide additional context while
71//! preserving the original cause.
72//!
73//! ## Guarantees
74//!
75//! - All memory accesses are bounds-checked
76//! - Register `x0` is always zero
77//! - The program counter always points to valid memory
78//! - State is not modified on error (operations are atomic)
79

83/// Base address for the UART MMIO region.
84const UART_BASE: u32 = 0x1000_0000;
85
86/// UART transmit register (write-only).
87/// Writing a byte here outputs it to stdout.
88const UART_TX: u32 = UART_BASE;
89
90/// UART receive register (read-only).
91/// Reading blocks until a byte is available from stdin.
92const UART_RX: u32 = UART_BASE;
93
94/// UART status register.
95/// Bit 0 (`TX_READY`) indicates the transmitter is ready.
96const UART_STATUS: u32 = UART_BASE + 0x04;
97
98/// UART transmitter ready flag.
99const TX_READY: u8 = 0b0000_0001;

101/// Errors that can occur in the virtual machine layer.
102///
103/// # Design
104///
105/// This error type represents failures at the lowest level of the emulator.
106///
107/// Higher layers (such as [`crate::cpu::CPUError`] and [`crate::risc_v::RiscVError`]) wrap this type
108/// to provide additional context.
109///
110/// # Variants
111///
112/// - [`VMError::ELFTooLarge`] — ELF segments exceed available RAM
113/// - [`VMError::InvalidSegment`] — ELF segment is malformed
114/// - [`VMError::InvalidRamSize`] — RAM size overflowed during calculation
115/// - [`VMError::MemoryOutOfBounds`] — invalid memory access
116/// - [`VMError::InvalidRegister`] — invalid register index
117/// - [`VMError::UnalignedPC`] — PC is not properly aligned
118/// - [`VMError::PCOverflow`] — PC arithmetic overflowed
119#[derive(Debug, Clone, PartialEq)]
120pub enum VMError {
121    ELFTooLarge,
122    InvalidSegment,
123    InvalidRamSize,
124    MemoryOutOfBounds(u32),
125    InvalidRegister(usize),
126    UnalignedPC(u32),
127    PCOverflow,
128}

130/// A simple RV32I virtual machine.
131///
132/// This struct represents the lowest-level abstraction of the emulator,
133/// providing memory, registers, and basic I/O primitives.
134///
135/// # Responsibilities
136///
137/// - Manage RAM and enforce memory safety
138/// - Provide register access (`x0..x31`)
139/// - Maintain the program counter (PC)
140/// - Expose memory-mapped I/O (e.g. UART)
141///
142/// # Guarantees
143///
144/// - Register `x0` is always zero
145/// - All memory accesses are bounds-checked
146/// - State is not modified on error
147///
148/// # Memory model
149///
150/// Memory is represented as a contiguous byte array. All accesses are
151/// little-endian, following the RISC-V specification.
152///
153/// # I/O
154///
155/// A minimal UART device is exposed via memory-mapped I/O:
156///
157/// - `0x1000_0000` — TX (write)
158/// - `0x1000_0000` — RX (read)
159/// - `0x1000_0004` — STATUS
160///
161/// # Examples
162///
163/// ```no_run
164/// use riscv::{RiscV, read_elf};
165///
166/// # let bytes: &[u8] = &[0; 4];
167/// # let elf = read_elf(bytes).unwrap();
168/// let mut vm = RiscV::new(elf, 1024).unwrap();
169/// ```
170#[derive(Debug)]
171pub struct VM {
172    ram: Vec<u8>,
173    registers: [u32; 32],
174    pc: u32,
175}
176
177impl VM {
178    /// Creates a new virtual machine from an ELF image.
179    ///
180    /// This method:
181    ///
182    /// - Allocates RAM
183    /// - Validates all ELF segments
184    /// - Loads segments into memory
185    /// - Initializes the program counter to the ELF entry point
186    ///
187    /// # Errors
188    ///
189    /// Returns [`VMError`] if:
190    ///
191    /// - The RAM size overflows
192    /// - A segment does not fit in memory
193    /// - A segment is malformed
194    ///
195    /// # Notes
196    ///
197    /// All segments are validated before any memory is allocated or modified.
198    /// This guarantees that the VM is never partially initialized.
199    pub fn new(elf_file: ElfImage, ram_length_kb: usize) -> Result<Self, VMError> {
200        let ram_size = ram_length_kb
201            .checked_mul(1024)
202            .ok_or(VMError::InvalidRamSize)?;
203
204        // Validate all segments before allocating anything
205        for seg in &elf_file.segments {
206            let seg_end = (seg.vaddr as usize)
207                .checked_add(seg.mem_size as usize)
208                .ok_or(VMError::ELFTooLarge)?;
209
210            if seg_end > ram_size {
211                return Err(VMError::ELFTooLarge);
212            }
213
214            // file_size can never be larger than mem_size (the ELF specification prohibits this)
215            if seg.data.len() > seg.mem_size as usize {
216                return Err(VMError::InvalidSegment);
217            }
218        }
219
220        let mut ram = vec![0u8; ram_size];
221
222        // Copies each PT_LOAD segment to its corresponding address in RAM.
223        // The bytes from (filesz .. memsz) are set to zero (BSS), as already ensured
224        // by vec![0u8; ram_size] above.
225        for seg in &elf_file.segments {
226            let start = seg.vaddr as usize;
227            let end = start + seg.data.len();
228            ram[start..end].copy_from_slice(&seg.data);
229        }
230
231        Ok(VM {
232            ram,
233            registers: [0u32; 32],
234            // The program counter starts at the entry point defined by the ELF, not at 0
235            // Original bug: the program counter was always set to 0
236            pc: elf_file.entry,
237        })
238    }

242    /// Reads a 32-bit little-endian value from memory.
243    ///
244    /// The value is read from the range `addr..addr+4`.
245    ///
246    /// # Errors
247    ///
248    /// Returns [`VMError::MemoryOutOfBounds`] if the access exceeds RAM bounds.
249    ///
250    /// # Guarantees
251    ///
252    /// The VM state is not modified if an error occurs.
253    pub fn read_u32(&self, addr: u32) -> Result<u32, VMError> {
254        let a = addr as usize;
255        if a + 4 > self.ram.len() {
256            return Err(VMError::MemoryOutOfBounds(addr));
257        }
258        Ok(u32::from_le_bytes(self.ram[a..a + 4].try_into().unwrap()))
259    }
260
261    /// Writes a 32-bit little-endian value to memory.
262    ///
263    /// The value is written to the range `addr..addr+4`.
264    ///
265    /// # Errors
266    ///
267    /// Returns [`VMError::MemoryOutOfBounds`] if the access exceeds RAM bounds.
268    ///
269    /// # Guarantees
270    ///
271    /// Memory is not modified if an error occurs.
272    pub fn write_u32(&mut self, addr: u32, value: u32) -> Result<(), VMError> {
273        let a = addr as usize;
274        if a + 4 > self.ram.len() {
275            return Err(VMError::MemoryOutOfBounds(addr));
276        }
277        self.ram[a..a + 4].copy_from_slice(&value.to_le_bytes());
278        Ok(())
279    }

281    /// Reads a byte from memory or a memory-mapped device.
282    ///
283    /// This method supports both regular RAM access and memory-mapped I/O.
284    ///
285    /// # Memory-mapped I/O
286    ///
287    /// The following addresses have special behavior:
288    ///
289    /// - `UART_STATUS` — returns the transmitter status flag
290    /// - `UART_RX` — reads a byte from standard input (blocking)
291    ///
292    /// All other addresses are treated as normal RAM.
293    ///
294    /// # Errors
295    ///
296    /// Returns [`VMError::MemoryOutOfBounds`] if the address is outside RAM
297    /// and does not correspond to a valid memory-mapped register.
298    ///
299    /// # Guarantees
300    ///
301    /// The VM state is not modified if an error occurs.
302    pub fn read_u8(&self, addr: u32) -> Result<u8, VMError> {
303        match addr {
304            UART_STATUS => return Ok(TX_READY),
305            UART_RX => {
306                let mut buf = [0u8];
307                std::io::stdin().read_exact(&mut buf).unwrap();
308                return Ok(buf[0]);
309            }
310            _ => {}
311        }
312
313        let a = addr as usize;
314        if a >= self.ram.len() {
315            return Err(VMError::MemoryOutOfBounds(addr));
316        }
317        Ok(self.ram[a])
318    }
319
320    /// Writes a byte to memory or a memory-mapped device.
321    ///
322    /// This method supports both regular RAM access and memory-mapped I/O.
323    ///
324    /// # Memory-mapped I/O
325    ///
326    /// The following addresses have special behavior:
327    ///
328    /// - `UART_TX` — writes a character to standard output
329    ///
330    /// All other addresses are treated as normal RAM.
331    ///
332    /// # Errors
333    ///
334    /// Returns [`VMError::MemoryOutOfBounds`] if the address is outside RAM
335    /// and does not correspond to a valid memory-mapped register.
336    ///
337    /// # Guarantees
338    ///
339    /// Memory is not modified if an error occurs.
340    pub fn write_u8(&mut self, addr: u32, value: u8) -> Result<(), VMError> {
341        if addr == UART_TX {
342            print!("{}", value as char);
343            return Ok(());
344        }
345
346        let a = addr as usize;
347        if a >= self.ram.len() {
348            return Err(VMError::MemoryOutOfBounds(addr));
349        }
350        self.ram[a] = value;
351        Ok(())
352    }

378    /// The value is written to the range `addr..addr+2`.

399    /// Returns the value of a general-purpose register.
400    ///
401    /// Registers are indexed from `0` to `31`, corresponding to `x0..x31`
402    /// in the RISC-V specification.
403    ///
404    /// # Errors
405    ///
406    /// Returns [`VMError::InvalidRegister`] if `index >= 32`.
407    ///
408    /// # Guarantees
409    ///
410    /// The VM state is not modified.
411    pub fn get_x(&self, index: usize) -> Result<u32, VMError> {
412        if index >= self.registers.len() {
413            return Err(VMError::InvalidRegister(index));
414        }
415        Ok(self.registers[index])
416    }
417
418    /// Sets the value of a general-purpose register.
419    ///
420    /// Registers are indexed from `0` to `31`, corresponding to `x0..x31`.
421    ///
422    /// Writes to register `x0` are ignored, as it is hardwired to zero
423    /// in the RISC-V specification.
424    ///
425    /// # Errors
426    ///
427    /// Returns [`VMError::InvalidRegister`] if `index >= 32`.
428    ///
429    /// # Guarantees
430    ///
431    /// - Register `x0` is never modified
432    /// - No state is modified if an error occurs
433    pub fn set_x(&mut self, index: usize, value: u32) -> Result<(), VMError> {
434        if index >= self.registers.len() {
435            return Err(VMError::InvalidRegister(index));
436        }
437
438        // x0 is hardwired to zero in RISC-V
439        if index == 0 {
440            return Ok(()); // ignore writes
441        }
442
443        self.registers[index] = value;
444        Ok(())
445    }

447    /// Returns the total size of RAM in bytes.
448    ///
449    /// This method provides read-only access to the VM memory size without
450    /// exposing the internal memory representation.
451    ///
452    /// # Usage
453    ///
454    /// Typically used by the CPU layer to initialize the stack pointer to
455    /// the top of RAM.
456    ///
457    /// # Guarantees
458    ///
459    /// The returned value always reflects the full allocated RAM size.
460    pub fn ram_size(&self) -> usize {
461        self.ram.len()
462    }

464    /// Returns the current program counter (PC).
465    ///
466    /// The program counter represents the address of the next instruction
467    /// to be executed.
468    ///
469    /// # Guarantees
470    ///
471    /// The VM state is not modified.
472    pub fn get_pc(&self) -> u32 {
473        self.pc
474    }
475
476    /// Sets the program counter (PC).
477    ///
478    /// The program counter determines the address of the next instruction
479    /// to be executed.
480    ///
481    /// This method ensures that the value points to a valid address in RAM,
482    /// but does not enforce instruction alignment.
483    ///
484    /// # Errors
485    ///
486    /// Returns [`VMError::MemoryOutOfBounds`] if `value` is outside RAM.
487    ///
488    /// # Guarantees
489    ///
490    /// - The PC is not modified if an error occurs
491    /// - Instruction alignment is not validated at this level
492    ///
493    /// # Notes
494    ///
495    /// Alignment and control-flow correctness are the responsibility of the CPU layer.
496    pub fn set_pc(&mut self, value: u32) -> Result<(), VMError> {
497        let addr = value as usize;
498
499        if addr >= self.ram.len() {
500            return Err(VMError::MemoryOutOfBounds(value));
501        }
502
503        self.pc = value;
504        Ok(())
505    }
