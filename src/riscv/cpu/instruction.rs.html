1//! RV32IM instruction decoding and intermediate representation.
2//!
3//! This module defines the [`Instruction`] enum, a canonical, fully-decoded
4//! representation of a RISC-V RV32IM instruction.
5//!
6//! # Overview
7//!
8//! The decode stage transforms a raw 32-bit instruction word into an
9//! [`Instruction`] value. This representation is **lossless at the semantic
10//! level** and eliminates all encoding-specific concerns (bitfields,
11//! immediate reconstruction, masking).
12//!
13//! The execution stage (`CPU::execute`) operates exclusively on this enum,
14//! treating it as a stable intermediate representation (IR) of the program.
15//!
16//! # Guarantees
17//!
18//! Decoding enforces the following invariants:
19//!
20//! - Register indices (`rd`, `rs1`, `rs2`) are always in the range `0..32`
21//! - Immediates are fully reconstructed and sign-extended to `i32`
22//! - Shift amounts are masked to 5 bits (`0..31`)
23//! - Instruction variants map 1:1 to RV32IM operations
24//!
25//! As a result, the execution stage does not need to perform any additional
26//! decoding, masking, or sign-extension.
27//!
28//! # Design notes
29//!
30//! - The enum acts as a boundary between decoding and execution
31//! - All encoding-specific complexity is confined to the decode stage
32//! - Execution logic can assume well-formed inputs
33//!
34//! # See also
35//!
36//! - [`crate::cpu::CPU`]
37//! - [`crate::vm::VM`]
38

41/// A decoded RV32IM instruction.
42///
43/// This enum represents the result of the decode stage for the
44/// RISC-V Unprivileged ISA (RV32I). Each variant corresponds to a
45/// single architectural instruction, with all operands and immediates
46/// fully decoded and normalized for execution.
47///
48/// The execution stage (`CPU::execute`) consumes this representation
49/// to update architectural state (register file, memory, and program counter).
50///
51/// # Execution model
52///
53/// Instruction semantics are fully defined by each variant.
54///
55/// - Register writes occur only when specified by the instruction
56/// - Memory access occurs only in load/store instructions
57/// - Program counter updates occur only in control-flow instructions
58/// - The CPU advances PC by 4 for non-control-flow instructions
59///
60/// No additional or implicit side effects exist.
61///
62/// # Field conventions
63///
64/// | Field   | Type  | Meaning                                               |
65/// |---------|-------|-------------------------------------------------------|
66/// | `rd`    | `u8`  | Destination register index (0–31)                     |
67/// | `rs1`   | `u8`  | Source register 1 index (0–31)                        |
68/// | `rs2`   | `u8`  | Source register 2 index (0–31)                        |
69/// | `imm`   | `i32` | Immediate value, sign-extended to 32 bits             |
70/// | `shamt` | `u8`  | Shift amount, masked to bits `[4:0]`                  |
71///
72/// # Architectural invariants
73///
74/// - Register `x0` is hardwired to zero; writes to `rd = 0` are discarded.
75/// - All arithmetic uses two's-complement wrapping semantics.
76///
77/// # Specification
78///
79/// Semantics follow the RISC-V Unprivileged ISA specification (RV32I).
80///
81/// # Coverage
82///
83/// This enum covers the complete RV32IM base ISA, including:
84/// - Integer arithmetic and logical instructions
85/// - Control flow (branches and jumps)
86/// - Memory access (loads and stores)
87/// - System (`ECALL`, `EBREAK`) and memory-ordering (`FENCE`, `FENCE.I`)
88#[derive(Debug, Clone, Copy, PartialEq, Eq)]
89pub enum Instruction {
90    // =========================================================
91    // R-type
92    // =========================================================
93    /// Integer addition.
94    ///
95    /// # Semantics
96    /// `x[rd] = x[rs1] + x[rs2]`
97    ///
98    /// # Effects
99    /// - Writes the 32-bit wrapping sum to register `rd`.
100    ///
101    /// # Notes
102    /// - Overflow wraps silently (two's-complement modular arithmetic).
103    Add { rd: u8, rs1: u8, rs2: u8 },
104
105    /// Integer subtraction.
106    ///
107    /// # Semantics
108    /// `x[rd] = x[rs1] - x[rs2]`
109    ///
110    /// # Effects
111    /// - Writes the 32-bit wrapping difference to register `rd`.
112    ///
113    /// # Notes
114    /// - Underflow wraps silently (two's-complement modular arithmetic).
115    Sub { rd: u8, rs1: u8, rs2: u8 },
116
117    /// Shift left logical (register).
118    ///
119    /// # Semantics
120    /// `x[rd] = x[rs1] << (x[rs2] & 0x1f)`
121    ///
122    /// # Effects
123    /// - Shifts `x[rs1]` left by the **lower 5 bits** of `x[rs2]`, filling
124    ///   vacated bits with zeros, and writes the result to register `rd`.
125    ///
126    /// # Notes
127    /// - Only the five least-significant bits of `rs2` are used as the shift
128    ///   amount; the upper bits are ignored.
129    Sll { rd: u8, rs1: u8, rs2: u8 },
130
131    /// Set less than (signed).
132    ///
133    /// # Semantics
134    /// `x[rd] = (x[rs1] as i32 < x[rs2] as i32) as u32`
135    ///
136    /// # Effects
137    /// - Writes `1` to register `rd` if `x[rs1]` is strictly less than
138    ///   `x[rs2]` under **signed** interpretation; writes `0` otherwise.
139    Slt { rd: u8, rs1: u8, rs2: u8 },
140
141    /// Set less than (unsigned).
142    ///
143    /// # Semantics
144    /// `x[rd] = (x[rs1] < x[rs2]) as u32`
145    ///
146    /// # Effects
147    /// - Writes `1` to register `rd` if `x[rs1]` is strictly less than
148    ///   `x[rs2]` under **unsigned** interpretation; writes `0` otherwise.
149    ///
150    /// # Notes
151    /// - Differs from [`Instruction::Slt`] when operands have their sign bit set, e.g.
152    ///   `0xFFFF_FFFF` is `-1` (signed) but `u32::MAX` (unsigned).
153    Sltu { rd: u8, rs1: u8, rs2: u8 },
154
155    /// Bitwise XOR (register).
156    ///
157    /// # Semantics
158    /// `x[rd] = x[rs1] ^ x[rs2]`
159    ///
160    /// # Effects
161    /// - Writes the bitwise exclusive-OR of `x[rs1]` and `x[rs2]` to
162    ///   register `rd`.
163    Xor { rd: u8, rs1: u8, rs2: u8 },
164
165    /// Shift right logical (register).
166    ///
167    /// # Semantics
168    /// `x[rd] = x[rs1] >> (x[rs2] & 0x1f)`
169    ///
170    /// # Effects
171    /// - Shifts `x[rs1]` right by the **lower 5 bits** of `x[rs2]`, filling
172    ///   vacated bits with **zeros**, and writes the result to register `rd`.
173    ///
174    /// # Notes
175    /// - Logical shift: the most-significant bit is filled with `0`
176    ///   regardless of the sign of `x[rs1]`.  See [`Instruction::Sra`] for
177    ///   sign-preserving right shift.
178    Srl { rd: u8, rs1: u8, rs2: u8 },
179
180    /// Shift right arithmetic (register).
181    ///
182    /// # Semantics
183    /// `x[rd] = ((x[rs1] as i32) >> (x[rs2] & 0x1f)) as u32`
184    ///
185    /// # Effects
186    /// - Shifts `x[rs1]` right by the **lower 5 bits** of `x[rs2]`, filling
187    ///   vacated bits with the **sign bit** of `x[rs1]`, and writes the
188    ///   result to register `rd`.
189    ///
190    /// # Notes
191    /// - Arithmetic shift: preserves the sign of the original value.
192    ///   Contrast with [`Instruction::Srl`], which always fills with zeros.
193    Sra { rd: u8, rs1: u8, rs2: u8 },
194
195    /// Bitwise OR (register).
196    ///
197    /// # Semantics
198    /// `x[rd] = x[rs1] | x[rs2]`
199    ///
200    /// # Effects
201    /// - Writes the bitwise OR of `x[rs1]` and `x[rs2]` to register `rd`.
202    Or { rd: u8, rs1: u8, rs2: u8 },
203
204    /// Bitwise AND (register).
205    ///
206    /// # Semantics
207    /// `x[rd] = x[rs1] & x[rs2]`
208    ///
209    /// # Effects
210    /// - Writes the bitwise AND of `x[rs1]` and `x[rs2]` to register `rd`.
211    And { rd: u8, rs1: u8, rs2: u8 },
212

214    // I-type ALU
215    // =========================================================
216    /// Add immediate.
217    ///
218    /// # Semantics
219    /// `x[rd] = x[rs1] + imm`
220    ///
221    /// # Effects
222    /// - Writes the 32-bit wrapping sum of `x[rs1]` and the sign-extended
223    ///   12-bit immediate to register `rd`.
224    ///
225    /// # Notes
226    /// - The canonical no-op `NOP` is encoded as `ADDI x0, x0, 0`.
227    /// - Overflow wraps silently (two's-complement modular arithmetic).
228    Addi { rd: u8, rs1: u8, imm: i32 },
229
230    /// Set less than immediate (signed).
231    ///
232    /// # Semantics
233    /// `x[rd] = (x[rs1] as i32 < imm) as u32`
234    ///
235    /// # Effects
236    /// - Writes `1` to register `rd` if `x[rs1]` is strictly less than the
237    ///   sign-extended immediate under **signed** comparison; writes `0`
238    ///   otherwise.
239    Slti { rd: u8, rs1: u8, imm: i32 },
240
241    /// Set less than immediate (unsigned).
242    ///
243    /// # Semantics
244    /// `x[rd] = (x[rs1] < imm as u32) as u32`
245    ///
246    /// # Effects
247    /// - Writes `1` to register `rd` if `x[rs1]` is strictly less than the
248    ///   immediate reinterpreted as an **unsigned** 32-bit value; writes `0`
249    ///   otherwise.
250    ///
251    /// # Notes
252    /// - The immediate is first sign-extended to 32 bits and then compared
253    ///   as an unsigned value, so `SLTIU rd, rs1, 1` tests whether `x[rs1]`
254    ///   is equal to zero.
255    Sltiu { rd: u8, rs1: u8, imm: i32 },
256
257    /// XOR immediate.
258    ///
259    /// # Semantics
260    /// `x[rd] = x[rs1] ^ imm as u32`
261    ///
262    /// # Effects
263    /// - Writes the bitwise XOR of `x[rs1]` and the sign-extended immediate
264    ///   to register `rd`.
265    ///
266    /// # Notes
267    /// - `XORI rd, rs1, -1` (all-ones immediate) produces the bitwise NOT
268    ///   of `x[rs1]`.
269    Xori { rd: u8, rs1: u8, imm: i32 },
270
271    /// OR immediate.
272    ///
273    /// # Semantics
274    /// `x[rd] = x[rs1] | imm as u32`
275    ///
276    /// # Effects
277    /// - Writes the bitwise OR of `x[rs1]` and the sign-extended immediate
278    ///   to register `rd`.
279    Ori { rd: u8, rs1: u8, imm: i32 },
280
281    /// AND immediate.
282    ///
283    /// # Semantics
284    /// `x[rd] = x[rs1] & imm as u32`
285    ///
286    /// # Effects
287    /// - Writes the bitwise AND of `x[rs1]` and the sign-extended immediate
288    ///   to register `rd`.
289    Andi { rd: u8, rs1: u8, imm: i32 },
290
291    /// Shift left logical immediate.
292    ///
293    /// # Semantics
294    /// `x[rd] = x[rs1] << (shamt & 0x1f)`
295    ///
296    /// # Effects
297    /// - Shifts `x[rs1]` left by `shamt` bit positions, filling vacated
298    ///   bits with zeros, and writes the result to register `rd`.
299    ///
300    /// # Notes
301    /// - `shamt` is pre-masked to bits `[4:0]` during decoding; values
302    ///   above 31 are not representable in the encoding.
303    Slli { rd: u8, rs1: u8, shamt: u8 },
304
305    /// Shift right logical immediate.
306    ///
307    /// # Semantics
308    /// `x[rd] = x[rs1] >> (shamt & 0x1f)`
309    ///
310    /// # Effects
311    /// - Shifts `x[rs1]` right by `shamt` bit positions, filling vacated
312    ///   bits with **zeros**, and writes the result to register `rd`.
313    Srli { rd: u8, rs1: u8, shamt: u8 },
314
315    /// Shift right arithmetic immediate.
316    ///
317    /// # Semantics
318    /// `x[rd] = ((x[rs1] as i32) >> (shamt & 0x1f)) as u32`
319    ///
320    /// # Effects
321    /// - Shifts `x[rs1]` right by `shamt` bit positions, filling vacated
322    ///   bits with the **sign bit** of `x[rs1]`, and writes the result to
323    ///   register `rd`.
324    Srai { rd: u8, rs1: u8, shamt: u8 },

330    ///
331    /// # Semantics
332    /// `x[rd] = sign_extend(mem[x[rs1] + imm][7:0])`
333    ///
334    /// # Effects
335    /// - Computes the effective address `x[rs1] + imm` using wrapping

337    /// - Reads 1 byte from memory at that address.
338    /// - Sign-extends the byte to 32 bits and writes the result to register
339    ///   `rd`.

343    ///
344    /// # Semantics
345    /// `x[rd] = sign_extend(mem[x[rs1] + imm][15:0])`
346    ///
347    /// # Effects
348    /// - Computes the effective address `x[rs1] + imm` using wrapping

350    /// - Reads 2 bytes (a halfword) from memory at that address.
351    /// - Sign-extends the 16-bit value to 32 bits and writes the result to
352    ///   register `rd`.
353    Lh { rd: u8, rs1: u8, imm: i32 },

355    /// Load word.
356    ///
357    /// # Semantics
358    /// `x[rd] = mem[x[rs1] + imm][31:0]`
359    ///
360    /// # Effects
361    /// - Computes the effective address `x[rs1] + imm` using wrapping
362    ///   addition.

364    /// - Writes the 32-bit value directly to register `rd`; no
365    ///   sign-extension is needed.
366    Lw { rd: u8, rs1: u8, imm: i32 },
367
368    /// Load byte (zero-extended).
369    ///
370    /// # Semantics
371    /// `x[rd] = zero_extend(mem[x[rs1] + imm][7:0])`
372    ///
373    /// # Effects
374    /// - Computes the effective address `x[rs1] + imm` using wrapping
375    ///   addition.
376    /// - Reads 1 byte from memory at that address.
377    /// - **Zero-extends** the byte to 32 bits (upper 24 bits are cleared)
378    ///   and writes the result to register `rd`.
379    ///
380    /// # Notes
381    /// - Differs from [`Instruction::Lb`] only in the extension strategy: `LBU` always
382    ///   produces a non-negative value in `rd`.
383    Lbu { rd: u8, rs1: u8, imm: i32 },
384
385    /// Load halfword (zero-extended).
386    ///
387    /// # Semantics
388    /// `x[rd] = zero_extend(mem[x[rs1] + imm][15:0])`
389    ///
390    /// # Effects
391    /// - Computes the effective address `x[rs1] + imm` using wrapping
392    ///   addition.
393    /// - Reads 2 bytes from memory at that address.
394    /// - **Zero-extends** the halfword to 32 bits (upper 16 bits are
395    ///   cleared) and writes the result to register `rd`.
396    ///
397    /// # Notes
398    /// - Differs from [`Instruction::Lh`] only in the extension strategy: `LHU` always

446    /// Branch if equal.
447    ///
448    /// # Semantics
449    /// `if x[rs1] == x[rs2] { PC = PC + imm }`
450    ///
451    /// # Effects
452    /// - Compares `x[rs1]` and `x[rs2]` for equality.
453    /// - **Taken**: sets the PC to `PC + imm` (PC-relative, wrapping).
454    /// - **Not taken**: PC is left unmodified; the caller advances it by 4.
455    /// - Does not access memory.
456    /// - Does not modify any register.
457    Beq { rs1: u8, rs2: u8, imm: i32 },
458

460    ///
461    /// # Semantics
462    /// `if x[rs1] != x[rs2] { PC = PC + imm }`
463    ///
464    /// # Effects
465    /// - Compares `x[rs1]` and `x[rs2]` for inequality.
466    /// - **Taken**: sets the PC to `PC + imm` (PC-relative, wrapping).
467    /// - **Not taken**: PC is left unmodified.
468    /// - Does not access memory.
469    /// - Does not modify any register.
470    Bne { rs1: u8, rs2: u8, imm: i32 },
471

473    ///
474    /// # Semantics
475    /// `if (x[rs1] as i32) < (x[rs2] as i32) { PC = PC + imm }`

477    /// # Effects
478    /// - Compares `x[rs1]` and `x[rs2]` as **signed** 32-bit integers.
479    /// - **Taken**: sets the PC to `PC + imm` (PC-relative, wrapping).
480    /// - **Not taken**: PC is left unmodified.
481    /// - Does not access memory.
482    /// - Does not modify any register.
483    Blt { rs1: u8, rs2: u8, imm: i32 },

494    /// - Does not access memory.
495    /// - Does not modify any register.

505    /// - **Taken**: sets the PC to `PC + imm` (PC-relative, wrapping).
506    /// - **Not taken**: PC is left unmodified.
507    /// - Does not access memory.
508    /// - Does not modify any register.

516    /// # Effects

518    /// - **Taken**: sets the PC to `PC + imm` (PC-relative, wrapping).
519    /// - **Not taken**: PC is left unmodified.

521    /// - Does not modify any register.

560    // =========================================================
561    /// Jump and link.
562    ///
563    /// # Semantics
564    /// `x[rd] = PC + 4; PC = PC + imm`
565    ///
566    /// # Effects
567    /// - Saves the **return address** (`PC + 4`) to register `rd`.
568    /// - Jumps unconditionally to `PC + imm` (PC-relative, wrapping).
569    /// - Does not access memory.
570    ///
571    /// # Notes
572    /// - When `rd = x0` the return address is discarded; this encodes an
573    ///   unconditional jump without a link.
574    /// - The 20-bit immediate allows a ±1 MiB PC-relative range.
575    Jal { rd: u8, imm: i32 },
576
577    /// Jump and link register.
578    ///
579    /// # Semantics
580    /// `x[rd] = PC + 4; PC = (x[rs1] + imm) & !1`
581    ///
582    /// # Effects
583    /// - Saves the **return address** (`PC + 4`) to register `rd`.
584    /// - Computes the target as `x[rs1] + imm` (wrapping) and **clears
585    ///   bit 0** to ensure 2-byte alignment, then sets the PC to that
586    ///   target.
587    /// - Does not access memory.
588    ///
589    /// # Notes
590    /// - Clearing bit 0 is mandated by the RISC-V spec to support
591    ///   compressed-instruction alignment; it is always applied even if the
592    ///   C extension is absent.
593    /// - `JALR x0, x1, 0` is the canonical function-return sequence.
594    Jalr { rd: u8, rs1: u8, imm: i32 },

598    // =========================================================

604    ///
605    /// # Effects
606    /// - Reads the **syscall number** from register `x17` (`a7`).
607    /// - Dispatches to the appropriate handler:

611    /// - Does not modify the PC directly (the CPU stops or errors before
612    ///   the normal PC-advance step).
613    ///
614    /// # Errors

617    Ecall,
618
619    /// Breakpoint.
620    ///
621    /// # Semantics
622    /// Signals the execution environment that a breakpoint has been reached.
623    ///
624    /// # Effects

627    /// - Does not modify any register, memory, or the PC.

629    /// # Notes

632    Ebreak,

637    /// Memory fence.
638    ///
639    /// # Semantics
640    /// Orders memory I/O accesses visible to other hardware threads or

646    ///
647    /// # Notes
648    /// - On a real multi-core RISC-V implementation, `FENCE` enforces

655    ///
656    /// # Semantics

677    ///
678    /// # Effects
679    /// - Multiplies `x[rs1]` and `x[rs2]` as 32-bit integers.
680    /// - Writes the least-significant 32 bits of the 64-bit product to `rd`.

721    ///
722    /// # Effects

752    /// - Writes the result to `rd`.
753    ///
754    /// # Special cases

756    /// - If `x[rs1] == INT32_MIN` and `x[rs2] == -1`, writes `0`.
757    Rem { rd: u8, rs1: u8, rs2: u8 },

773impl Instruction {
774    fn fmt_asm(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
775        use Instruction::*;
776
777        match *self {
778            // =========================
779            // R-type
780            // =========================
781            Add { rd, rs1, rs2 } => r(f, "add", rd, rs1, rs2),
782            Sub { rd, rs1, rs2 } => r(f, "sub", rd, rs1, rs2),
783            Sll { rd, rs1, rs2 } => r(f, "sll", rd, rs1, rs2),
784            Slt { rd, rs1, rs2 } => r(f, "slt", rd, rs1, rs2),
785            Sltu { rd, rs1, rs2 } => r(f, "sltu", rd, rs1, rs2),
786            Xor { rd, rs1, rs2 } => r(f, "xor", rd, rs1, rs2),
787            Srl { rd, rs1, rs2 } => r(f, "srl", rd, rs1, rs2),
788            Sra { rd, rs1, rs2 } => r(f, "sra", rd, rs1, rs2),
789            Or { rd, rs1, rs2 } => r(f, "or", rd, rs1, rs2),
790            And { rd, rs1, rs2 } => r(f, "and", rd, rs1, rs2),
791
792            // =========================
793            // RV32M
794            // =========================
795            Mul { rd, rs1, rs2 } => r(f, "mul", rd, rs1, rs2),
796            Mulh { rd, rs1, rs2 } => r(f, "mulh", rd, rs1, rs2),
797            Mulhsu { rd, rs1, rs2 } => r(f, "mulhsu", rd, rs1, rs2),
798            Mulhu { rd, rs1, rs2 } => r(f, "mulhu", rd, rs1, rs2),
799            Div { rd, rs1, rs2 } => r(f, "div", rd, rs1, rs2),
800            Divu { rd, rs1, rs2 } => r(f, "divu", rd, rs1, rs2),
801            Rem { rd, rs1, rs2 } => r(f, "rem", rd, rs1, rs2),
802            Remu { rd, rs1, rs2 } => r(f, "remu", rd, rs1, rs2),
803
804            // =========================
805            // I-type ALU
806            // =========================
807            Addi { rd, rs1, imm } => i(f, "addi", rd, rs1, imm),
808            Slti { rd, rs1, imm } => i(f, "slti", rd, rs1, imm),
809            Sltiu { rd, rs1, imm } => i(f, "sltiu", rd, rs1, imm),
810            Xori { rd, rs1, imm } => i(f, "xori", rd, rs1, imm),
811            Ori { rd, rs1, imm } => i(f, "ori", rd, rs1, imm),
812            Andi { rd, rs1, imm } => i(f, "andi", rd, rs1, imm),
813
814            Slli { rd, rs1, shamt } => ish(f, "slli", rd, rs1, shamt),
815            Srli { rd, rs1, shamt } => ish(f, "srli", rd, rs1, shamt),
816            Srai { rd, rs1, shamt } => ish(f, "srai", rd, rs1, shamt),
817
818            // =========================
819            // Loads
820            // =========================
821            Lb { rd, rs1, imm } => load(f, "lb", rd, rs1, imm),
822            Lh { rd, rs1, imm } => load(f, "lh", rd, rs1, imm),
823            Lw { rd, rs1, imm } => load(f, "lw", rd, rs1, imm),
824            Lbu { rd, rs1, imm } => load(f, "lbu", rd, rs1, imm),
825            Lhu { rd, rs1, imm } => load(f, "lhu", rd, rs1, imm),
826
827            // =========================
828            // Stores
829            // =========================
830            Sb { rs1, rs2, imm } => store(f, "sb", rs1, rs2, imm),
831            Sh { rs1, rs2, imm } => store(f, "sh", rs1, rs2, imm),
832            Sw { rs1, rs2, imm } => store(f, "sw", rs1, rs2, imm),
833
834            // =========================
835            // Branches
836            // =========================
837            Beq { rs1, rs2, imm } => branch(f, "beq", rs1, rs2, imm),
838            Bne { rs1, rs2, imm } => branch(f, "bne", rs1, rs2, imm),
839            Blt { rs1, rs2, imm } => branch(f, "blt", rs1, rs2, imm),
840            Bge { rs1, rs2, imm } => branch(f, "bge", rs1, rs2, imm),
841            Bltu { rs1, rs2, imm } => branch(f, "bltu", rs1, rs2, imm),
842            Bgeu { rs1, rs2, imm } => branch(f, "bgeu", rs1, rs2, imm),
843
844            // =========================
845            // U-type
846            // =========================
847            Lui { rd, imm } => u(f, "lui", rd, imm),
848            Auipc { rd, imm } => u(f, "auipc", rd, imm),
849
850            // =========================
851            // Jumps
852            // =========================
853            Jal { rd, imm } => u(f, "jal", rd, imm),
854            Jalr { rd, rs1, imm } => jalr(f, rd, rs1, imm),
855
856            // =========================
857            // System
858            // =========================
859            Ecall => f.write_str("ecall"),
860            Ebreak => f.write_str("ebreak"),
861
862            // =========================
863            // Fence
864            // =========================
865            Fence => f.write_str("fence"),
866            FenceI => f.write_str("fence.i"),
867        }
868    }
869}

871impl fmt::Display for Instruction {
872    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
873        self.fmt_asm(f)
874    }
875}

877// helpers

879fn r(f: &mut fmt::Formatter<'_>, op: &str, rd: u8, rs1: u8, rs2: u8) -> fmt::Result {
880    write!(f, "{op} x{rd}, x{rs1}, x{rs2}")
881}

883#[inline(always)]
884fn i(f: &mut fmt::Formatter<'_>, op: &str, rd: u8, rs1: u8, imm: i32) -> fmt::Result {
885    write!(f, "{op} x{rd}, x{rs1}, {imm}")
886}
887
888#[inline(always)]
889fn ish(f: &mut fmt::Formatter<'_>, op: &str, rd: u8, rs1: u8, shamt: u8) -> fmt::Result {
890    write!(f, "{op} x{rd}, x{rs1}, {shamt}")
891}

893#[inline(always)]
894fn load(f: &mut fmt::Formatter<'_>, op: &str, rd: u8, base: u8, off: i32) -> fmt::Result {
895    write!(f, "{op} x{rd}, {off}(x{base})")
896}
897
898#[inline(always)]
899fn store(f: &mut fmt::Formatter<'_>, op: &str, base: u8, src: u8, off: i32) -> fmt::Result {
900    write!(f, "{op} x{src}, {off}(x{base})")
901}
902
903#[inline(always)]
904fn branch(f: &mut fmt::Formatter<'_>, op: &str, rs1: u8, rs2: u8, off: i32) -> fmt::Result {
905    write!(f, "{op} x{rs1}, x{rs2}, {off}")
906}
907
908#[inline(always)]
909fn u(f: &mut fmt::Formatter<'_>, op: &str, rd: u8, imm: i32) -> fmt::Result {

911}
