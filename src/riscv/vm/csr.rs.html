1//! # Control and Status Registers (CSRs)
2//!
3//! This module implements the RISC-V CSR file for machine-mode (`M-mode`)
4//! operation, covering the subset required to run firmware such as FreeRTOS
5//! on an RV32IM processor.
6//!
7//! ## Implemented registers
8//!
9//! | Address | Name        | Description                          |
10//! |---------|-------------|--------------------------------------|
11//! | `0x300` | `mstatus`   | Machine status                       |
12//! | `0x301` | `misa`      | ISA and extensions (read-only)       |
13//! | `0x304` | `mie`       | Machine interrupt enable             |
14//! | `0x305` | `mtvec`     | Trap-handler base address            |
15//! | `0x340` | `mscratch`  | Scratch register for trap handlers   |
16//! | `0x341` | `mepc`      | Machine exception program counter    |
17//! | `0x342` | `mcause`    | Machine trap cause                   |
18//! | `0x344` | `mip`       | Machine interrupt pending            |
19//! | `0xC00` | `cycle`     | Cycle counter (low 32 bits)          |
20//! | `0xC80` | `cycleh`    | Cycle counter (high 32 bits)         |
21//! | `0xC02` | `instret`   | Instructions retired (low 32 bits)   |
22//! | `0xC82` | `instreth`  | Instructions retired (high 32 bits)  |
23//! | `0xF14` | `mhartid`   | Hardware thread ID (read-only)       |
24//!
25//! ## Write semantics
26//!
27//! Certain registers enforce constraints on write:
28//!
29//! - **`mstatus`** — WPRI (reserved) bits are masked to zero via [`MSTATUS_MASK`]
30//! - **`mtvec`** — bit 1 is reserved in direct mode and is always cleared
31//! - **`mepc`** — bits 1:0 are always zero (instruction alignment)
32//! - **`mip`** — the `MTIP` bit (bit 7) is read-only from software; it is
33//!   driven exclusively by the CLINT hardware
34//! - **`misa`**, **`mhartid`**, **`cycle`**, **`instret`** — read-only;
35//!   write calls targeting these addresses return `false` and leave state unchanged
36
37/// Writable bit mask for [`Csrs::mip`].
38///
39/// Bit 7 (`MTIP`) is driven by the CLINT and is not writable by software.
40/// All other bits may be written freely.
41pub const MIP_WRITABLE: u32 = !(1 << 7);
42
43/// Write mask for [`Csrs::mstatus`].
44///
45/// Clears all WPRI (reserved) bits as required by the RISC-V privileged spec.
46/// Only the following bits are preserved after a write:
47///
48/// - Bit 3 — `MIE` (Machine Interrupt Enable)
49/// - Bit 7 — `MPIE` (Machine Previous Interrupt Enable)
50/// - Bits 11:12 — `MPP` (Machine Previous Privilege)
51pub const MSTATUS_MASK: u32 = 0x0001_9988;
52
53/// The CSR file for a single RISC-V M-mode hart.
54///
55/// Holds all control and status registers needed for trap handling,
56/// interrupt management, and performance counting.
57///
58/// # Usage
59///
60/// Access is always performed through [`Csrs::read`] and [`Csrs::write`],
61/// which enforce the masking and read-only semantics defined by the spec.
62/// Direct field access is available for internal emulator use (e.g. the CLINT
63/// setting `mip.MTIP`).
64///
65/// # Example
66///
67/// ```rust
68/// use riscv::vm::csr::Csrs;
69/// let mut csrs = Csrs::default();
70///
71/// // Enable machine timer interrupts (MTIE = bit 7 of mie)
72/// csrs.write(0x304, 1 << 7);
73/// assert_eq!(csrs.read(0x304), Some(1 << 7));
74///
75/// // mhartid is read-only — writes are silently ignored
76/// assert!(!csrs.write(0xF14, 0xFF));
77/// assert_eq!(csrs.read(0xF14), Some(0));
78/// ```
79#[derive(Debug, Default)]
80pub struct Csrs {
81    /// Machine status register (`mstatus`, `0x300`).
82    ///
83    /// Controls global interrupt enable (`MIE`), previous interrupt state
84    /// (`MPIE`), and previous privilege mode (`MPP`). WPRI bits are always
85    /// masked to zero on write via [`MSTATUS_MASK`].
86    pub mstatus: u32,
87
88    /// ISA and extensions register (`misa`, `0x301`). Read-only.
89    ///
90    /// Encodes the supported ISA. Writes are silently ignored.
91    pub misa: u32,
92
93    /// Machine interrupt-enable register (`mie`, `0x304`).
94    ///
95    /// Each bit enables a specific interrupt source. The relevant bits for
96    /// FreeRTOS are:
97    /// - Bit 7 — `MTIE` (Machine Timer Interrupt Enable)
98    pub mie: u32,
99
100    /// Trap-handler base address register (`mtvec`, `0x305`).
101    ///
102    /// Holds the base address of the trap vector. Bit 1 is reserved in
103    /// direct mode and is always cleared on write.
104    pub mtvec: u32,
105
106    /// Scratch register for machine trap handlers (`mscratch`, `0x340`).
107    ///
108    /// Conventionally used by the trap handler to save a register before
109    /// swapping in the kernel stack pointer.
110    pub mscratch: u32,
111
112    /// Machine exception program counter (`mepc`, `0x341`).
113    ///
114    /// Holds the PC of the instruction that caused the trap. Bits 1:0 are
115    /// always zero (instruction alignment is enforced on write).
116    pub mepc: u32,
117
118    /// Machine cause register (`mcause`, `0x342`).
119    ///
120    /// Encodes the reason for the most recent trap. The MSB distinguishes
121    /// interrupts (1) from exceptions (0).
122    pub mcause: u32,
123
124    /// Machine interrupt-pending register (`mip`, `0x344`).
125    ///
126    /// Reflects which interrupts are currently pending. Bit 7 (`MTIP`) is
127    /// set by the CLINT hardware and is read-only from software.
128    pub mip: u32,
129
130    /// Hardware thread ID (`mhartid`, `0xF14`). Read-only.
131    ///
132    /// Always `0` in a single-hart emulator. Writes are silently ignored.
133    pub mhartid: u32,
134
135    /// Cycle counter (`cycle`/`cycleh`, `0xC00`/`0xC80`). Read-only.
136    ///
137    /// Counts clock cycles since reset. Exposed as two 32-bit reads:
138    /// low half at `0xC00`, high half at `0xC80`.
139    pub cycle: u64,
140
141    /// Instructions-retired counter (`instret`/`instreth`, `0xC02`/`0xC82`). Read-only.
142    ///
143    /// Incremented by the emulator after each successfully executed instruction.
144    /// Exposed as two 32-bit reads: low half at `0xC02`, high half at `0xC82`.
145    pub instret: u64,
146}
147
148impl Csrs {
149    /// Reads a CSR by address.
150    ///
151    /// 64-bit counters (`cycle`, `instret`) are split across two addresses:
152    /// the base address returns the low 32 bits and base `+ 0x80` returns
153    /// the high 32 bits.
154    ///
155    /// # Parameters
156    ///
157    /// - `addr` — 12-bit CSR address as defined by the RISC-V privileged spec
158    ///
159    /// # Returns
160    ///
161    /// - `Some(value)` if the address maps to a supported register
162    /// - `None` if the address is not implemented
163    ///
164    /// # Supported addresses
165    ///
166    /// See the [module-level register table](self).
167    pub fn read(&self, addr: u16) -> Option<u32> {
168        match addr {
169            0x300 => Some(self.mstatus),
170            0x301 => Some(self.misa),
171            0x304 => Some(self.mie),
172            0x305 => Some(self.mtvec),
173            0x340 => Some(self.mscratch),
174            0x341 => Some(self.mepc),
175            0x342 => Some(self.mcause),
176            0x344 => Some(self.mip),
177            0xC00 => Some(self.cycle as u32),
178            0xC80 => Some((self.cycle >> 32) as u32),
179            0xC02 => Some(self.instret as u32),
180            0xC82 => Some((self.instret >> 32) as u32),
181            0xF14 => Some(self.mhartid),
182            _ => None,
183        }
184    }
185
186    /// Writes a value to a CSR by address.
187    ///
188    /// Write constraints are enforced automatically:
189    ///
190    /// - `mstatus` — WPRI bits are masked via [`MSTATUS_MASK`]
191    /// - `mtvec` — bit 1 is cleared
192    /// - `mepc` — bits 1:0 are cleared
193    /// - `mip` — bit 7 (`MTIP`) is preserved from the current hardware state
194    ///
195    /// Read-only registers (`misa`, `mhartid`, `cycle`, `instret`) are not
196    /// listed in the write match and will return `false`.
197    ///
198    /// # Parameters
199    ///
200    /// - `addr` — 12-bit CSR address
201    /// - `value` — value to write
202    ///
203    /// # Returns
204    ///
205    /// - `true` if the address maps to a writable register
206    /// - `false` if the address is not implemented or is read-only
207    ///
208    /// # Guarantees
209    ///
210    /// - If this method returns `false`, CSR state is unchanged
211    pub fn write(&mut self, addr: u16, value: u32) -> bool {
212        match addr {
213            0x300 => self.mstatus = value & MSTATUS_MASK,
214            0x304 => self.mie = value,
215            0x305 => self.mtvec = value & !0b10,
216            0x340 => self.mscratch = value,
217            0x341 => self.mepc = value & !0b11,
218            0x342 => self.mcause = value,
219            0x344 => self.mip = (self.mip & !MIP_WRITABLE) | (value & MIP_WRITABLE),
220            _ => return false,
221        }
222        true
223    }
224}
